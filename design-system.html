CSS Custom Properties

Todos os tokens importados via design-tokens.css. Use sempre os tokens, nunca valores hardcoded.

/* CORES */
--gold: #D9B855          --gold-light: #F2DEA0      --gold-grad: linear-gradient(135deg, ...)
--blue-deep: #010A26     --blue-royal: #00113F      --blue-grad: linear-gradient(160deg, ...)

/* TIPOGRAFIA */
--font-display: "Oswald"  --font-body: "Raleway"
--text-xs: 14px           --text-sm: 18px           --text-md: 22px
--text-lg: 28px           --text-xl: 34px           --text-2xl: 48px
--text-fluid-sm: clamp(38px, 5.5vw, 80px)

/* ESPAÇAMENTO */
--sp-1: 4px   --sp-2: 8px    --sp-3: 12px   --sp-4: 16px
--sp-5: 20px  --sp-6: 24px   --sp-8: 32px   --sp-10: 40px

/* BORDAS */
--radius-sm: 4px   --radius-md / --radius-lg / --radius-xl: 16px   --radius-pill: 999px
--border-gold: 1px solid var(--gold-30)   --border-white: 1px solid var(--white-20)

/* MOTION */
--transition-fast: 0.15s ease-out   --transition-base: 0.3s ease   --transition-slow: 0.75s ease

/* Z-INDEX */
--z-slide: 1   --z-ui: 100   --z-nav: 9998   --z-cursor: 9999
